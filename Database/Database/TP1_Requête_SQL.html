1. Afficher marque :

SELECT `marque`

FROM `voitures`

  1. Afficher marque et couleur :

SELECT `marque`, `couleur`

FROM `voitures`

  1. Afficher Immatriculation et couleur des voitures noires :

SELECT `immatriculation`, `couleur`

FROM `voitures`

WHERE `couleur` = 'noire'

  1. Afficher toutes les informations des clients qui habitent à Paris :

SELECT *

FROM `clients`

WHERE `ville` = 'paris'

  1. Afficher toutes les informations concernant les voitures ayant un kilométrage <10000

SELECT *

FROM `voitures`

WHERE `kilometrage` <10000

  1. Afficher les immatriculations des voitures diesel ayant un nombre de porte égal à 5 :

SELECT *

FROM `voitures`

WHERE `nbre_porte` = 5

  1. Afficher l’immatriculation, la marque, le modèle de toute les voitures Mercedes :

SELECT `immatriculation`, `marque`, `modele`

FROM `voitures`

WHERE `marque` = 'mercedes'
